# Consultas referentes a Lighting

from rdflib import Graph

# Definición de prefijos
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
"""

# Consultas SPARQL
queries_lighting = [
    # 1. Listar todos los elementos de alumbrado y su tipo
    prefixes + """
    SELECT ?lighting ?type
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasType ?type .
    }
    LIMIT 10
    """,
    # 2. Elementos de alumbrado promovidos por un promotor específico y su posición
    prefixes + """
    SELECT ?lighting ?position
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasPromoter "AG OBRAS Y EQUIPAMIENTOS" .
        ?lighting tm:hasStreet ?position .
    }
    LIMIT 10
    """,
    # 3. Listar los motivos de alumbrado utilizados en cada distrito
    prefixes + """
    SELECT ?district ?districtName ?motive
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasMotive ?motive .

        ?lighting tm:hasDistrict ?district .
        ?district owl:sameAs ?districtName .
    }
    LIMIT 10
    """,
    # 4. Obtener información completa de un alumbrado
    prefixes + """
    SELECT ?property ?value
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting ?property ?value .
    }
    LIMIT 10
    """,
    # 5. Listar todos los barrios y los elementos de alumbrado que pertenecen a ellos
    prefixes + """
    SELECT ?neighborhood ?neighborhoodName ?lighting
    WHERE {
        ?lighting a tm:Lighting .

        ?lighting tm:hasNeighborhood ?neighborhood .
        ?neighborhood owl:sameAs ?neighborhoodName .
    }
    LIMIT 10
    """,
    # 6. Alumbrados con longitud mayor que 500
    prefixes + """
    SELECT ?lighting ?length
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasLength ?length .
        FILTER(xsd:integer(?length) > 500)
    }
    LIMIT 10
    """,
    # 7. Promedio de longitud de los alumbrados en cada distrito
    prefixes + """
    SELECT ?district ?districtName (AVG(xsd:integer(?length)) AS ?avgLength)
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasLength ?length .

        ?lighting tm:hasDistrict ?district .
        ?district owl:sameAs ?districtName .
    }
    GROUP BY ?district ?districtName
    LIMIT 10
    """,
    # 8. Consultar todos los alumbrados con un motivo específico
    prefixes + """
    SELECT ?lighting ?district ?districtName ?neighborhood ?neighborhoodName ?type
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasMotive ?motive .
        FILTER(lcase(?motive) = "arcos compactos")
        ?lighting tm:hasType ?type .

        ?lighting tm:hasDistrict ?district .
        ?lighting tm:hasNeighborhood ?neighborhood .
        ?district owl:sameAs ?districtName .
        ?neighborhood owl:sameAs ?neighborhoodName .
    }
    LIMIT 10
    """
]

# Definición de prefijos
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
"""

# Consultas SPARQL
queries_museum = [
    # 1. Listar todos los museos y sus direcciones
    prefixes + """
    SELECT ?museum ?museum_link ?address
    WHERE {
        ?museum a tm:Museum .
        ?museum owl:sameAs ?museum_link .
        ?museum tm:hasAddress ?address .
    }
    LIMIT 15
    """,
    # 2. Consultar museos que están en un barrio específico
    prefixes + """
    SELECT ?museum ?museum_link ?neighborhood_link
    WHERE {
        ?museum a tm:Museum .
        ?museum owl:sameAs ?museum_link .
        ?museum tm:hasAddress ?address .
        ?address tm:hasNeighborhood tm:palacio .
        tm:palacio owl:sameAs ?neighborhood_link .
    }
    LIMIT 15
    """,
    # 3. Contar el número de museos en cada distrito
    prefixes + """
    SELECT ?district (COUNT(?museum) AS ?numMuseums)
    WHERE {
        ?museum a tm:Museum .
        ?museum tm:hasAddress ?address .
        ?address tm:hasDistrict ?district .
    }
    GROUP BY ?district
    LIMIT 15
    """,
    # 4. Encontrar museos accesibles que tienen transporte cercano
    prefixes + """
    SELECT ?museum ?museum_link
    WHERE {
        ?museum a tm:Museum .
        ?museum owl:sameAs ?museum_link .
        ?museum tm:hasAccessibility ?accessibility .
        ?museum tm:hasNearbyTransport ?transport .
        FILTER(?accessibility = "1")
    }
    LIMIT 15
    """,
    # 5. Obtener información completa de una dirección
    prefixes + """
    SELECT ?property ?value
    WHERE {
        tm:4398134475439 ?property ?value .
    }
    LIMIT 15
    """,
    # 6. Agrupar barrios por distrito y contar cuántos hay en cada uno
    prefixes + """
    SELECT ?district ?district_link (COUNT(?museum) AS ?nummuseum)
    WHERE {
        ?museum a tm:Museum .
        ?museum tm:hasAddress ?address .
        ?address tm:hasDistrict ?district .
        ?district owl:sameAs ?district_link .
    }
    GROUP BY ?district
    LIMIT 15
    """,
    # 7. Museos que están en una calle específica
    prefixes + """
    SELECT ?museum ?museum_link
    WHERE {
        ?museum a tm:Museum .
        ?museum owl:sameAs ?museum_link .
        ?museum tm:hasAddress ?address .
        ?address tm:hasStreet tm:serrano .
    }
    LIMIT 15
    """,
    # 8. Museos dentro de un radio de una ubicación (latitud, longitud)
    prefixes + """
    SELECT ?museum ?museum_link
    WHERE {
        ?museum a tm:Museum .
        ?museum owl:sameAs ?museum_link .
        ?museum tm:hasAddress ?address .
        ?address tm:hasLatitude ?lat .
        ?address tm:hasLongitude ?long .
        FILTER (
            (xsd:decimal(?lat) - 40.4168)*(xsd:decimal(?lat) - 40.4168) +
            (xsd:decimal(?long) - (-3.7038))*(xsd:decimal(?long) - (-3.7038)) <= 0.01
        )
    }
    LIMIT 15
    """,
    # 9. Museos que están en un barrio específico
    prefixes + """
    SELECT ?museum ?museum_link ?neighborhood_link
    WHERE {
        ?museum a tm:Museum .
        ?museum owl:sameAs ?museum_link .
        ?museum tm:hasAddress ?address .
        ?address tm:hasNeighborhood tm:universidad .
        tm:universidad owl:sameAs ?neighborhood_link .
    }
    LIMIT 15
    """
]

from rdflib import Graph

# Definición de prefijos
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
"""

# Consultas SPARQL
queries_monument = [
    # 1. Listar todos los monumentos y sus direcciones
    prefixes + """
    SELECT ?museum ?monument_link ?address
    WHERE {
        ?monument a tm:Monument .
        ?monument owl:sameAs ?monument_link .
        ?monument tm:hasAddress ?address .
    }
    LIMIT 15
    """,
    # 2. Consultar monumentos que están en un barrio específico
    prefixes + """
    SELECT ?monument ?monument_link ?neighborhood_link
    WHERE {
        ?monument a tm:Monument .
        ?monument owl:sameAs ?monument_link .
        ?monument tm:hasAddress ?address .
        ?address tm:hasNeighborhood tm:vallehermoso .
        tm:vallehermoso owl:sameAs ?neighborhood_link .
    }
    LIMIT 15
    """,
    # 3. Contar el número de monumentos en cada distrito
    prefixes + """
    SELECT ?district (COUNT(?monument) AS ?numMonuments)
    WHERE {
        ?monument a tm:Monument .
        ?monument tm:hasAddress ?address .
        ?address tm:hasDistrict ?district .
    }
    GROUP BY ?district
    LIMIT 15
    """,
    # 4. Obtener información completa de una dirección
    prefixes + """
    SELECT ?property ?value
    WHERE {
        tm:4395684477304 ?property ?value .
    }
    LIMIT 15
    """,
    # 5. Agrupar barrios por distrito y contar cuántos hay en cada uno
    prefixes + """
    SELECT ?district ?district_link (COUNT(?monument) AS ?nummonument)
    WHERE {
        ?monument a tm:Monument .
        ?monument tm:hasAddress ?address .
        ?address tm:hasDistrict ?district .
        ?district owl:sameAs ?district_link .
    }
    GROUP BY ?district
    LIMIT 15
    """,
    # 6. Monumentos que están en una calle específica
    prefixes + """
    SELECT ?monument ?street
    WHERE {
        ?monument a tm:Monument .

        ?monument tm:hasAddress ?address .
        ?address tm:hasStreet ?street .
    }
    LIMIT 15
    """,
    # 7. Monumentos dentro de un radio de una ubicación (latitud, longitud)
    prefixes + """
    SELECT ?monument ?monument_link
    WHERE {
        ?monument a tm:Monument .
        ?monument owl:sameAs ?monument_link .
        ?monument tm:hasAddress ?address .
        ?address tm:hasLatitude ?lat .
        ?address tm:hasLongitude ?long .
        FILTER (
            (xsd:decimal(?lat) - 40.44420091117868)*(xsd:decimal(?lat) - 40.44420091117868) +
            (xsd:decimal(?long) - (-3.712595448820414))*(xsd:decimal(?long) - (-3.712595448820414)) <= 0.01
        )
    }
    LIMIT 15
    """,
]

# Consultas SPARQL

# Definición de prefijos
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
"""

# Consultas SPARQL
queries_event = [
    # 1. Listar todos los eventos y sus direcciones
    prefixes + """
    SELECT ?event ?address
    WHERE {
        ?event a tm:Event .
        ?event tm:hasAddress ?address .
    }
    LIMIT 10
    """,
    # 2. Consultar eventos que están en un barrio específico, considerando equivalencias (owl:sameAs)
    prefixes + """
    SELECT ?event ?neighborhood_link
    WHERE {
        ?event a tm:Event .
        ?event tm:hasActivity ?activity .
        ?activity tm:takesPlaceAtInstallation ?installation .
        ?installation  tm:hasAddress ?address .
        ?address tm:hasNeighborhood ?neighborhood .
        ?neighborhood owl:sameAs ?neighborhood_link .
        FILTER(?neighborhood = tm:la-chopera)
    }
    LIMIT 10
    """,
    # 3. Contar el número de eventos en cada distrito (incluyendo equivalencias)
    prefixes + """
    SELECT ?district (COUNT(?event) AS ?numEvents)
    WHERE {
        ?event a tm:Event .
        ?event tm:hasActivity ?activity .
        ?activity tm:takesPlaceAtInstallation ?installation .
        ?installation  tm:hasAddress ?address .
        ?address tm:hasNeighborhood ?neighborhood .
        ?address tm:hasDistrict ?district .
        OPTIONAL { ?district owl:sameAs ?equivalentDistrict . }
    }
    GROUP BY ?district
    LIMIT 10
    """,
    # 4. Eventos gratuitos
    prefixes + """
    SELECT ?event
    WHERE {
        ?event a tm:Event .
        ?event tm:hasFreeAdmission ?freeadmision .
        FILTER(?freeadmision  = 1)
    }
    LIMIT 10
    """,
    # 5. Obtener información completa de una dirección
    prefixes + """
    SELECT ?title ?address
    WHERE {
       ?event a tm:Event .
        ?event tm:hasTitle ?title .
        ?event tm:hasActivity ?activity .
        ?activity tm:takesPlaceAtInstallation ?installation .
        ?installation  tm:hasAddress ?address .
    }
    LIMIT 10
    """,
    # 6. Agrupar barrios por distrito y contar cuántos hay en cada uno, considerando equivalencias
    prefixes + """
    SELECT ?district (COUNT(DISTINCT ?neighborhood) AS ?numNeighborhoods)
    WHERE {
        ?address tm:hasDistrict ?district .
        ?address tm:hasNeighborhood ?neighborhood .
        OPTIONAL { ?neighborhood owl:sameAs ?equivalentNeighborhood . }
    }
    GROUP BY ?district
    LIMIT 10
    """,
    # 7. Eventos en una calle específica
    prefixes + """
    SELECT ?event
    WHERE {
        ?event a tm:Event .
        ?event tm:hasActivity ?activity .
        ?activity tm:takesPlaceAtInstallation ?installation .
        ?installation  tm:hasAddress ?address .
        ?address tm:hasStreet tm:fuencarral .
    }
    LIMIT 10
    """,
    # 8. Eventos dentro de un radio de una ubicación (considerando latitud y longitud)
    prefixes + """
    SELECT ?event
    WHERE {
        ?event a tm:Event .
        ?event tm:hasActivity ?activity .
        ?activity tm:takesPlaceAtInstallation ?installation .
        ?installation tm:hasAddress ?address .
        ?address tm:hasLatitude ?lat .
        ?address tm:hasLongitude ?long .
        FILTER (
            (xsd:decimal(?lat) - 40.4168)*(xsd:decimal(?lat) - 40.4168) +
            (xsd:decimal(?long) - (-3.7038))*(xsd:decimal(?long) - (-3.7038)) <= 0.01
        )
    }
    LIMIT 10
    """,
    # 9. Consultar eventos en un barrio específico, con equivalencias
    prefixes + """
    SELECT ?event ?district_link
    WHERE {
        ?event a tm:Event .
        ?event tm:hasActivity ?activity .
        ?activity tm:takesPlaceAtInstallation ?installation .
        ?installation  tm:hasAddress ?address .
        ?address tm:hasNeighborhood ?neighborhood .
        ?address tm:hasDistrict ?district .
        ?district owl:sameAs ?district_link .
        FILTER(?district = tm:tetuan)
    }
    LIMIT 10
    """
]

    SELECT DISTINCT ?barrio ?museo ?museo_link ?monumento ?monument0_link
    WHERE {
        ?museo a tm:Museum .
        ?museo owl:sameAs ?museo_link .
        ?museo tm:hasAddress ?direccion .
        ?direccion tm:hasNeighborhood ?barrio .

        ?monumento a tm:Monument .
        ?monumento owl:sameAs ?monument0_link .
        ?monumento tm:hasAddress ?monumentodireccion .
        ?monumentodireccion tm:hasNeighborhood ?barrio .

    }
    LIMIT 15
    """,
    # 2. Museos y luces en la misma calle
    prefixes + """
    SELECT DISTINCT ?neighborhood ?museo ?museo_link ?luces
    WHERE {
      ?museo a tm:Museum .
      ?museo owl:sameAs ?museo_link .
      ?museo tm:hasAddress ?direccion .
      ?direccion tm:hasNeighborhood ?neighborhood .

      ?luces a tm:Lighting .
      ?luces tm:hasNeighborhood ?neighborhood .
    }
    LIMIT 15
    """,


    # 3. Contar recursos por tipo en cada barrio
    prefixes + """
    SELECT ?neighborhood ?neighborhood_link (COUNT(?museum) AS ?numMuseums) (COUNT(?lighting) AS ?numLightings)
    WHERE {
        OPTIONAL {
            ?museum a tm:Museum .
            ?museum tm:hasAddress ?museumAddress .
            ?museumAddress tm:hasNeighborhood ?neighborhood .
            ?neighborhood owl:sameAs ?neighborhood_link .
        }
        OPTIONAL {
            ?lighting a tm:Lighting .
            ?lighting tm:hasNeighborhood ?neighborhood .
        }
    }
    GROUP BY ?neighborhood
    LIMIT 10
    """,

    # 4. Museos con información de iluminación relacionada
    prefixes + """

        ?museumAddress tm:hasNeighborhood ?neighborhood .

        ?lighting a tm:Lighting .
        ?lighting tm:hasStreet ?street .
        ?lighting tm:hasNeighborhood ?neighborhood .
    }
    LIMIT 10
    """,

    # 5. Barrios con más museos y luces
